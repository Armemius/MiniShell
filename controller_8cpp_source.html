9std::string formatTime(const double duration) {
10 constexpr double NS_THRESHOLD = 1e-6;
11 constexpr double US_THRESHOLD = 1e-3;
12 constexpr double MS_THRESHOLD = 1;
13 constexpr double M_THRESHOLD = 60;
14 constexpr double H_THRESHOLD = 60 * M_THRESHOLD;
15 constexpr double D_THRESHOLD = 24 * H_THRESHOLD;
16
17 if (duration < NS_THRESHOLD) {
18 return fmt::format("{}ns", static_cast<int>(duration * 1e9));
19 }
20 if (duration < US_THRESHOLD) {
21 return fmt::format("{}us", static_cast<int>(duration * 1e6));
22 }
23 if (duration < MS_THRESHOLD) {
24 return fmt::format("{}ms", static_cast<int>(duration * 1e3));
25 }
26 if (duration > D_THRESHOLD) {
27 return fmt::format("{}D {}", static_cast<int>(duration / D_THRESHOLD),
28 formatTime(std::fmod(duration, D_THRESHOLD)));
29 }
30 if (duration > H_THRESHOLD) {
31 return fmt::format("{}H {}", static_cast<int>(duration / H_THRESHOLD),
32 formatTime(std::fmod(duration, H_THRESHOLD)));
33 }
34 if (duration > M_THRESHOLD) {
35 return fmt::format("{}m {}", static_cast<int>(duration / M_THRESHOLD),
36 formatTime(std::fmod(duration, M_THRESHOLD)));
37 }
38 return fmt::format("{}s", static_cast<int>(duration));
39}
